$y$ & 1 & 2 \\
\hline

$\Pr(Y=y)$ & $\frac{1}{2}$ & $\frac{1}{2}$ \\[0.1cm]
\hline
\end{tabular}

\begin{enumerate}[a)]
\item \quad \\[-1.45cm]
\begin{align*}
E(Y) &= \left(1\times\frac{1}{2}\right) + \left(2\times\frac{1}{2}\right) \\
&= \frac{1}{2} + \frac{2}{2} = 1.5 \text{ unique faces}.
\end{align*}
\item \quad \\[-1.45cm]
\begin{align*}
E(Y^2) &= \left(1^2\times\frac{1}{2}\right) + \left(2^2\times\frac{1}{2}\right) \\
&= \frac{1}{2} + \frac{4}{2} = 2.5.
\end{align*}
\end{enumerate}

\framebox[0.5\textwidth]{
\begin{minipage}[t]{0.46\textwidth}
\subsection*{Question 2}
\begin{enumerate}[a)]
\item \quad \\[-1.45cm]
\begin{align*}
Var(Y) &= E(Y^2) - [E(Y)]^2\\
&= 2.5 - (1.5)^2\\
&= 2.5 - 2.25 \\
&= 0.25 \text{ unique faces$^2$.}
\end{align*}
\item \quad \\[-1.45cm]
\begin{align*}
Sd(Y) &= \sqrt{Var(Y)} \\
&= \sqrt{0.25} \\
&= 0.5 \text{ unique faces.}
\end{align*}
\end{enumerate}

&= \frac{2}{36} + \frac{6}{36} + \frac{12}{36} + \frac{20}{36} + \frac{30}{36} + \frac{42}{36} + \frac{40}{36} + \frac{36}{36} + \frac{30}{36} + \frac{22}{36} + \frac{12}{36} \\[0.2cm]

\end{align*}
\end{enumerate}

\subsection*{Question 3 continued}
\begin{enumerate}[a)]
\item[c)] \quad\\[-1.45cm]
\begin{align*}
E(X^2) &= \left(2^2\times\frac{1}{36}\right) + \left(3^2\times\frac{2}{36}\right) + \left(4^2\times\frac{3}{36}\right) + \left(5^2\times\frac{4}{36}\right) + \left(6^2\times\frac{5}{36}\right) + \left(7^2\times\frac{6}{36}\right) + \\

&= \left(4\times\frac{1}{36}\right) + \left(9\times\frac{2}{36}\right) + \left(16\times\frac{3}{36}\right) + \left(25\times\frac{4}{36}\right) + \left(36\times\frac{5}{36}\right) + \left(49\times\frac{6}{36}\right) + \\

\begin{minipage}[t]{0.47\textwidth}
\begin{enumerate}[a)]
\item[c)] If $X$ and $Y$ were independent then the joint distribution would be\\[-0.3cm]
\begin{adjustwidth}{-0.5cm}{0cm}
\begin{tabular}{c|c|cc|c|}
\multicolumn{2}{c}{} & \multicolumn{2}{c}{$X$} & \multicolumn{1}{c}{}\\

\item[d)] \quad\\[-1.45cm]
\begin{align*}
\Pr(X = 1 \,|\, Y=0) &= \frac{\Pr(X = 1 \cap Y=0)}{\Pr(Y=0)} \\
&=\frac{0.2}{0.6} = 0.33.\\
\end{align*}
\item[e)] \quad\\[-1.45cm]
\begin{align*}
E(X) &= 0(0.5) + 1(0.5) = 0.5.\\[0.3cm]
E(X^2) &= (0^2)(0.5) + (1^2)(0.5) = 0.5.\\[0.3cm]
Var(X) &= E(X^2) - [E(X)^2] \\&= 0.5 - (0.5)^2 = 0.25.\\[0.3cm]
Sd(x) &= \sqrt{Var(X)} = \sqrt{0.25} = 0.5.
\end{align*}
\end{enumerate}
