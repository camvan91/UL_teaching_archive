\subsection*{Question 1}
\begin{enumerate}[a)]
\item $X \sim \text{Binomial}(n=20,p=0.1)$. \\[-0.5cm]
\begin{align*}
\Pr(X=2) &= \binom{20}{2}\,(0.1^2)\,(0.9^{18}) = 0.2852.\\[0.5cm]
\lambda = n\,p &= 20(0.1) = 2\\[0.1cm]
\Pr(X=2) &\approx \frac{2^2}{2!}\,e^{-2} = 0.2707.\\[-0.2cm]
\end{align*}
\item $X \sim \text{Binomial}(n=100,p=0.02)$. \\[-0.5cm]
\begin{align*}
\Pr(X=5) &= \binom{100}{5}\,(0.02^5)\,(0.98^{95}) = 0.0353.\\[0.5cm]
\lambda = n\,p &= 100(0.02) = 2\\[0.1cm]
\Pr(X=5) &\approx \frac{2^5}{5!}\,e^{-2} = 0.0361.\\[-0.2cm]
\end{align*}
\end{enumerate}
\end{minipage}\hspace{0.055\textwidth}
\begin{minipage}[t]{0.47\textwidth}
\quad
\begin{enumerate}[a)]
\item[c)] $X \sim \text{Binomial}(n=1000,p=0.005)$. \\[-0.5cm]
\begin{align*}
\Pr(X=3) &= \binom{1000}{3}\,(0.005^3)\,(0.995^{997})\\ &= 0.1403.\\[0.5cm]
\lambda = n\,p &= 1000(0.005) = 5\\[0.1cm]
\Pr(X=3) &\approx \frac{5^3}{3!}\,e^{-5} =  0.1404.\\[-0.2cm]
\end{align*}
\item[d)]  $X \sim \text{Binomial}(n=10000,p=0.0001)$. \\[-0.5cm]
\begin{align*}
\Pr(X=1) &= \binom{10000}{1}\,(0.0001^1)\,(0.9999^{9999})\\ &= 0.3679.\\[0.5cm]
\lambda = n\,p &= 10000(0.0001) = 1\\[0.1cm]
\Pr(X=1) &\approx \frac{1^1}{1!}\,e^{-1} = 0.3679.\\[-0.2cm]
\end{align*}
\end{enumerate}

\subsection*{Question 2}
Here we have $X \sim \text{Poisson}(\lambda=2)$.\\[0.5cm]
$\Rightarrow$ $\Pr(X=x) = \frac{\lambda^x}{x\,!} \,e^{-\lambda} = \frac{2^x}{x\,!} \,e^{-2}$ in one hour.\\[-0.1cm]
\begin{enumerate}[a)]
\item \quad\\[-1.45cm]
\begin{align*}
\Pr(X=6) = \frac{2^6}{6\,!} \,e^{-2} = 0.0120.\\[-0.2cm]
\end{align*}
\item \quad\\[-1.45cm]
\begin{align*}
\Pr(X<3) &= \Pr(X \le 2) = p(0) + p(1) + p(2) \\[0.2cm]
&= \frac{2^0}{0\,!} \,e^{-2} + \frac{2^1}{1\,!} \,e^{-2} + \frac{2^2}{2\,!} \,e^{-2} \\[0.2cm]
&= 0.1353 + 0.2707 + 0.2707\\
&= 0.6767.\\[-0.2cm]
\end{align*}
\item Average is $2(2) = 4$ per two hours. \quad\\[-0.5cm]
\begin{align*}
\Rightarrow \Pr(X=0) = \frac{4^0}{0\,!} \,e^{-4} = 0.0183.\\[-0.2cm]
\end{align*}
\end{enumerate}
\end{minipage}\hspace{0.055\textwidth}
\begin{minipage}[t]{0.47\textwidth}
\quad\\[-1cm]
\begin{enumerate}[a)]
\item[d)] Again the average is 4 per two hours. \quad\\[-0.5cm]
\begin{align*}
&\Pr(X>4) = 1 - \Pr(X \leq 4) \\
&\quad= 1 - [p(0)+p(1)+(2)+p(3)+p(4)]\\[0.2cm]
&= 1 - \left(\frac{4^0}{0\,!} \,e^{-4} + \frac{4^1}{1\,!} \,e^{-4} + \frac{4^2}{2\,!} \,e^{-4} + \right. \\
& \qquad\qquad\qquad\qquad\qquad\left.\frac{4^3}{3\,!} \,e^{-4} + \frac{4^4}{4\,!} \,e^{-4}\right) \\[0.2cm]
&= 1 - (0.0183 + 0.0733 + 0.1465 + \\
&\qquad\qquad\qquad\qquad\qquad0.1954 + 0.1954)\\[0.1cm]
&= 1-0.6289=0.3711.
\end{align*}
\item[e)] The average is $2(0.5) = 1$ per 0.5 hours. \quad\\[-0.5cm]
\begin{align*}
\Pr(X\ge1) &= 1 - \Pr(X = 0) \\[0.2cm]
&= 1 - \frac{1^0}{0\,!} \,e^{-1}\\[0.2cm]
&= 1 - 0.3679 = 0.6321.
\end{align*}
\item[f)] $E(X) = \lambda = 2$ emails per hour.\\[0.4cm]
$Sd(X) = \sqrt{Var(X)} = \sqrt{\lambda} = \sqrt{2} = 1.414$ emails per hour.
\end{enumerate}
\end{minipage}

\subsection*{Question 3}
Same as Q2 but now using the Poisson tables. We must rework the questions in terms of \emph{greater than or equal to} probabilities.
\begin{enumerate}[a)]
\item \quad\\[-1.45cm]
\begin{align*}
\Pr(X=6) &= \Pr(X \ge 6) - \Pr(X \ge 7) \\
&= 0.0166 - 0.0045 = 0.0121.\\[-0.2cm]
\end{align*}
\item \quad\\[-1.45cm]
\begin{align*}
\Pr(X<3) &= 1 - \Pr(X \ge 3) \\
&= 1 - 0.3233 = 0.6767.\\[-0.2cm]
\end{align*}
\item Here $\lambda = 2(2) = 4$. \quad\\[-0.5cm]
\begin{align*}
\Pr(X=0) &= \Pr(X \ge 0) - \Pr(X \ge 1) \\
&= 1.000 - 0.9817= 0.0183.\\[-0.2cm]
\end{align*}
\end{enumerate}
\end{minipage}\hspace{0.055\textwidth}
\begin{minipage}[t]{0.47\textwidth}
\begin{enumerate}[a)]
\item[d)] Again $\lambda = 4$. \quad\\[-0.5cm]
\begin{align*}
\Pr(X>4) &= \Pr(X \ge 5) = 0.3712.\\[-0.2cm]
\end{align*}
\item[e)] Here $\lambda = 2(0.5) = 1$. \quad\\[-0.5cm]
\begin{align*}
\Pr(X\ge1) &= 0.6321.
\end{align*}
We can see that these are the same as above apart from small differences due to rounding.
\end{enumerate}
\end{minipage}
