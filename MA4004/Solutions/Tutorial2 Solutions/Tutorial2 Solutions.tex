\subsection*{Question 2}
\begin{enumerate}[a)]
\item \quad \\[-1.45cm]
\begin{align*}
\Pr(W \cup M) &= \Pr(W) + \Pr(M) - \Pr(W\cap M) \\
&= 0.7 + 0.2 - 0.1 \quad = 0.8.
\end{align*}
\item \quad \\[-1.45cm]
\begin{align*}
\Pr(W^c \cap M^c) &= 1 - \Pr(W \cup M) \\
&= 1 - 0.8\quad = 0.2.
\end{align*}
\item $W$ and $M$ are \emph{not} mutually exlusive since $\Pr(W\cap M) \ne 0$.
\item \quad \\[-1.45cm]
\begin{align*}
\Pr(W) \times \Pr(M) &= 0.7 \times 0.2 \\ &= 0.14 \\
&\ne \Pr(W \cap M)  = 0.1.\\[0.3cm]
\Rightarrow& \text{ \emph{not} independent}.
\end{align*}
\end{enumerate}
\end{minipage}}\vspace{0.03\textwidth}

\subsection*{Question 3}
\begin{enumerate}[a)]
\item \quad \\[-1.45cm]
\begin{align*}
\Pr(A \cup B) &= \Pr(A) + \Pr(B) - \Pr(A\cap B) \\
0.75 &= 0.45 + 0.6 - \Pr(A\cap B) \\
0.75 &= 1.05 - \Pr(A\cap B) \\
0.75 - 1.05 &= - \Pr(A\cap B) \\
- 0.3 &= - \Pr(A\cap B) \\[0.3cm]
\Rightarrow \Pr(A\cap B) &= 0.3.
\end{align*}
\item \quad \\[-1.45cm]
\begin{align*}
\Pr(A) \times \Pr(B) &= 0.45 \times 0.6 \\ &= 0.27 \\
&\ne \Pr(A \cap B)  = 0.3.\\[0.3cm]
\Rightarrow& \text{ \emph{not} independent}.
\end{align*}
\end{enumerate}
\end{minipage}\hspace{0.055\textwidth}
\begin{minipage}[t]{0.47\textwidth}
\begin{enumerate}
\item[c)] $\Pr(A^c) = 1 - \Pr(A) = 1 - 0.45 = 0.55.$\\[0.2cm]
$\Pr(B^c) = 1 - \Pr(B) = 1 - 0.6 = 0.4.$
\item[d)]  Note that $(A \cap B)^c = A^c \cup B^c$. \\[-0.3cm]
\begin{align*}
\Rightarrow \Pr(A^c \cup B^c) &= 1- \Pr(A \cap B)\\
&= 1 - 0.3 \\
&= 0.7.
\end{align*}
\end{enumerate}
\end{minipage}
